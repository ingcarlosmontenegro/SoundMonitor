\chapter{Objetivos}

\section{Objetivo general}
\begin{itemize}
    \item Desarrollar un sistema que permita la clasificación y análisis estadístico de las fuentes de ruido en entornos urbanos aplicando modelos de aprendizaje automático.
\end{itemize}
